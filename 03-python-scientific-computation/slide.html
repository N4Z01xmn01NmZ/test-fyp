Data types

                            
                                #include <stdio.h>
    
                                int main() {
                                    bool boo = true; // Stores true or false
                                    printf("boo value is %d with size of %d byte", boo, sizeof(bool));

                                    char cha = 'A'; // Stores a single byte
                                    printf("cha value is %d with size of %d byte", boo, sizeof(char));

                                    int num = 45; // Stores 4 bytes interger value
                                    printf("num value is %d with size of %d bytes", boo, sizeof(int));

                                    float flo = 3.14f; // Stores a floating point value
                                    printf("flo value is %2.3f %+.0e with size of %d bytes", boo, sizeof(float));
                                }
